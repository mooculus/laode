\begin{exercise} \label{c4.2.2c}
Find a $2\times 2$ matrix that reflects vectors in the $(x,y)$ plane across
the line $x=y$.

\begin{solution}
The map $L_A$ that reflects vectors across the line $y=x$ is
$(x,y) \rightarrow (y,x)$.  The matrix is
\[
A = \mattwo{0}{1}{1}{0}.
\]

\end{solution}
\end{exercise}
