\begin{exercise} \label{c10.2.6}
Let $A$ be an $n\times n$ matrix.  Suppose that 
\[
A^2 + A + I_n = 0.
\]
Prove that $A$ is invertible.

\begin{solution}

We are given $A^2 + A + I_n = 0$.  Therefore, $I_n = -A^2 - A =
A(-A - I_n)$.  Thus, $A^{-1} = -A - I_n$ exists.

\end{solution}
\end{exercise}
