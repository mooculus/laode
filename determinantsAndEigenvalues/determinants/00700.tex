\begin{exercise} \label{c10.1.b7a}
Compute the cofactor matrices $A_{13}, A_{22}, A_{21}$ when 
$A = \left( \begin{array}{rrr}
 3 & 2 & -4\\
 0 & 1 & 5\\
 0 & 0 & 6\end{array} \right)$.

\begin{solution}
\ans
$A_{13} = \mattwo{0}{1}{0}{0}$;
$A_{22} = \mattwo{3}{-4}{0}{6}$;
$A_{21} = \mattwo{2}{-4}{0}{6}$.

\end{solution}
\end{exercise}
